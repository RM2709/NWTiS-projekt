
// minimalno 20 aerodroma za preuzimanje
SELECT * FROM AERODROMI_LETOVI;

//  minimalno 50.000 preuzetih polazaka s aerodroma (bez ponavljanja istih)
SELECT COUNT(DISTINCT (ESTDEPARTUREAIRPORT, FIRSTSEEN)) AS BROJ_PREUZETIH_POLAZAKA FROM LETOVI_POLASCI;

// broj preuzetih podataka po danima za sve aerodrome pojedinačno i sortirano po aerodromu i danu
// minimalno 10 dana u slijedu u cijelosti za koje su preuzeti podaci polazaka za sve aerodrome
SELECT ESTDEPARTUREAIRPORT, CAST(TIMESTAMP(FIRSTSEEN) AS DATE) AS DATUM, COUNT(*) AS BROJ_POLAZAKA FROM LETOVI_POLASCI GROUP BY ESTDEPARTUREAIRPORT, DATUM ORDER BY ESTDEPARTUREAIRPORT, DATUM;

// broj preuzetih podataka po danima za sve aerodrome sortirano po danu
SELECT CAST(TIMESTAMP(FIRSTSEEN) AS DATE) AS DATUM, COUNT(*) AS BROJ_POLAZAKA FROM LETOVI_POLASCI GROUP BY DATUM ORDER BY DATUM;

// broj preuzetih podataka po danima za odabrani aerodrom sortirano po danu npr. LDZA
SELECT ESTDEPARTUREAIRPORT, CAST(TIMESTAMP(FIRSTSEEN) AS DATE) AS DATUM, COUNT(*) AS BROJ_POLAZAKA FROM LETOVI_POLASCI WHERE ESTDEPARTUREAIRPORT='LDZA' GROUP BY ESTDEPARTUREAIRPORT, DATUM ORDER BY ESTDEPARTUREAIRPORT, DATUM;

// broj preuzetih podataka za sve aerodrome pojedinačno sortirano po aerodromu na odabrani dan npr. 01.01.2023
SELECT ESTDEPARTUREAIRPORT, CAST(TIMESTAMP(FIRSTSEEN) AS DATE) AS DATUM, COUNT(*) AS BROJ_POLAZAKA FROM LETOVI_POLASCI WHERE CAST(TIMESTAMP(FIRSTSEEN) AS DATE)='2023-01-07' GROUP BY ESTDEPARTUREAIRPORT, DATUM ORDER BY ESTDEPARTUREAIRPORT, DATUM;
